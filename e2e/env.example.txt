# E2E 테스트 환경 설정
# 이 파일을 .env로 복사하여 사용하세요: cp env.example.txt .env

# 프론트엔드 URL
E2E_BASE_URL=http://localhost:3000

# 백엔드 API URL
E2E_BACKEND_URL=http://localhost:8000

# 헤드리스 모드 (true: 브라우저 숨김, false: 브라우저 표시)
E2E_HEADLESS=true

# 느린 모드 (디버깅용, 각 동작 후 1초 대기)
E2E_SLOW_MODE=false

# 스크린샷 저장 디렉토리
E2E_SCREENSHOT_DIR=screenshots

# 테스트 사용자 정보 (인증 활성화 후 사용)
E2E_TEST_USERNAME=testuser
E2E_TEST_PASSWORD=TestPassword123!

# 관리자 테스트 계정
E2E_ADMIN_USERNAME=admin
E2E_ADMIN_PASSWORD=AdminPassword123!
